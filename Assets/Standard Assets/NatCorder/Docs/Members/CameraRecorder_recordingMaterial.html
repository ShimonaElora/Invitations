CameraRecorder.recordingMaterial

public Material recordingMaterial;

Description

Material to use for recording the camera view.

This is useful for applying effects while recording. Set to null to record without a material. 

void RecordGreyscale () {
    // Load greyscale shader
    var greyscaleMaterial = new Material(Shader.Find("greyscaleShader"));
    // Start recording
    NatCorder.StartRecording(Container.MP4, VideoFormat.Screen, AudioFormat.None, OnRecording);
    var videoRecorder = CameraRecorder.Create(Camera.main);
    videoRecorder.recordingMaterial = greyscaleMaterial;
}
